十大排序算法——Java语言版

1.冒泡排序（Bubble Sort）

public static void bubble_sort(int[] arr, int len) {
    int i, j, temp;
    for (i = 0; i < len - 1; i++)
        for (j = 0; j < len - 1 - i; j++)
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
}
  • arr[]：需要排序的数组。
  • len：数组长度。

2.选择排序（Selection Sort）

public static void selection_sort(int[] arr, int len) {
    int i, j, temp;
    for (i = 0 ; i < len - 1 ; i++) {
        int min = i;
        for (j = i + 1; j < len; j++) {
            if (arr[j] < arr[min])
                min = j;
        }
        temp = arr[min];
        arr[min] = arr[i];
        arr[i] = temp;
    }
}
  • arr[]：需要排序的数组。
  • len：数组长度。

3.插入排序（Insertion Sort）

public static void insertion_sort(int[] arr, int len) {
    int i, j, key;
    for (i = 1; i < len; i++) {
        key = arr[i];
        j = i - 1;
        while((j >= 0) && (arr[j] > key)) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}
  • arr[]：需要排序的数组。
  • len：数组长度。

4.希尔排序（Shell Sort）

public static void shell_sort(int[] arr, int len) {
    int gap, i, j;
    int temp;
    for (gap = len >> 1; gap > 0; gap >>= 1)
        for (i = gap; i < len; i++) {
            temp = arr[i];
            for (j = i - gap; j >= 0 && arr[j] > temp; j -= gap)
                arr[j + gap] = arr[j];
            arr[j + gap] = temp;
        }
}
  • arr[]：需要排序的数组。
  • len：数组长度。
  • *>>*：右移一位（二进制） 如： 00001010（10） >> 等于 00000101（5） 这里是除2的作用。

5.归并排序（Merge Sort）

public static void merge_sort(int[] arr, int start, int end) {
    if(start < end) {
        int mid = (start + end) / 2;
        merge_sort(arr, start, mid);
        merge_sort(arr, mid + 1, end);
        merge(arr, start, mid, end);
    }
}

public static void merge(int[] arr, int start, int mid, int end) {
    int[] temp = new int[end - start + 1];
    int l = start, r = mid + 1, j = 0;
    while (l <= mid && r <= end) {
        if(arr[l] < arr[r]) {
            temp[j++] = arr[l++];
        }else{
            temp[j++] = arr[r++];
        }
    }
    while (l <= mid) {
        temp[j++] = arr[l++];
    }
    while (r <= end) {
        temp[j++] = arr[r++];
    }
    for (int i = 0; i < temp.length; i++) {
        arr[i + start] = temp[i];
    }
}
  • *arr：需要排序数组的地址。
  • start：开始下标。
  • mid：中间下标。
  • end：结束下标。

6.快速排序（Quick Sort）

public static void quick_sort(int arr[], int start, int end) {
    if (start < end) {
        int index = partition(arr, start, end);
        quick_sort(arr, start, index - 1);
        quick_sort(arr, index + 1, end);
    }
}

public static int partition(int arr[], int start, int end) {
    int temp = arr[start];
    int li = start, ri = end;
    while (li < ri) {
        while (li < ri && arr[ri] > temp)
            ri--;
        if (li < ri) {
            arr[li] = arr[ri];
            li++;
        }
        while (li < ri && arr[li] < temp)
            li++;
        if (li < ri) {
            arr[ri] = arr[li];
            ri--;
        }
    }
    arr[li] = temp;
    return li;
}
  • arr[]：需要排序的数组。
  • start：开始下标。
  • index：中间下标。
  • end：结束下标。

7.堆排序（Heap Sort）

public static void heap_sort(int arr[], int len) {
    for(int i = len; i > 1; i --) {
        heap(arr, i);
    }
}

public static void heap(int arr[], int len) {
    int dad = len / 2 - 1, son = 2 * dad + 1, temp;
    while (dad >= 0) {
        if (son + 1 <= len - 1 && arr[son] < arr[son + 1])
            son++;
        if (arr[dad] < arr[son]) {
            temp = arr[dad];
            arr[dad] = arr[son];
            arr[son] = temp;
        }
        dad--;
        son = 2 * dad + 1;
    }
    temp = arr[0];
    arr[0] = arr[len - 1];
    arr[len - 1] = temp;
}
  • arr[]：需要排序的数组。
  • len：数组长度。
  • dad：最后一个父节点。
  • son：该父节点下的首个子节点。
  • if (son + 1 <= len - 1 && arr[son] < arr[son + 1])：判断是否有两个子节点若有则在其中寻找最大子节点。

8.计数排序（Counting Sort）

public static void counting_Sort(int arr[], int len) {
    int max = arr[0], min = arr[0], i, j = 0;
    for (i = 0; i < len; i++) {
        if (arr[i] < min)
            min = arr[i];
        if (arr[i] > max)
            max = arr[i];
    }
    int new_len = max - min + 1;
    int[] conunting_arr = new int[new_len];
    for (i = 0; i < new_len; i++) {
        conunting_arr[i] = 0;
    }
    for (i = 0; i < len; i++) {
        conunting_arr[arr[i] - min]++;
    }
    for (i = 0; i < new_len; i++) {
        int index = conunting_arr[i];
        while (index != 0) {
            arr[j] = i + min;
            index--;
            j++;
        }
    }
}
  • arr[]：需要排序的数组。
  • len：数组长度。
  • conunting_arr：计数数组。
  • new_len：计数数组长度。

9.桶排序（Bucket Sort）

public static void bucket_sort(int arr[], int len) {
    int max = 0, j = 0;
    for (int i = 0; i < len; i++) {
        if (arr[i] > max)
            max = arr[i];
    }
    max += 1;
    int[] bucket = new int[max];
    for (int i = 0; i < len; i++)
        bucket[arr[i]]++;
    for (int i = 0; i < max; i++)
        while ((bucket[i]--) > 0)
            arr[j++] = i;
}
  • arr[]：需要排序的数组。
  • len：数组长度。
  • int* bucket = (int*)malloc(sizeof(int) * max)：动态开辟桶数量。
  • 12~13行：遍历数组,将元素放到对应桶中。
  • 14~16行：把桶内元素放回数组中。

10.基数排序（Radix Sort）

public static void radix_sort(int arr[], int len) {
    int level = 0;
    int max = arr[0], i;
    for (i = 0; i < len; i++) {
        if (arr[i] > max)
            max = arr[i];
    }
    for (i = max; i > 0; i /= 10) {
        level++;
    }
    for (i = 0; i < level; i++) {
        radix(arr, len, i);
    }
}

public static void radix(int arr[], int len, int level) {
    int[] bucket =new int[10];
    int[] temp = new int[len];
    int i, j;
    int flag = (int)Math.pow(10, level);
    for (i = 0; i < len; i++) {
        bucket[arr[i] / flag % 10]++;
    }
    bucket[0]--;
    for (i = 1; i < 10; i++) {
        bucket[i] += bucket[i - 1];
    }
    for (i = len - 1; i >= 0; i--) {
        temp[bucket[arr[i] / flag % 10]] = arr[i];
        bucket[arr[i] / flag % 10]--;
    }
    for (i = 0; i < len; i++) {
        arr[i] = temp[i];
    }
}
  • arr[]：需要排序的数组。
  • len：数组长度。
  • bucket：排序桶。
  • temp：临时数组。
